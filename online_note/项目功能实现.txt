3. 数据库脚本 sql/schema.sql

-- 创建数据库
CREATE DATABASE IF NOT EXISTS message_board_db;
USE message_board_db;

-- 留言表
CREATE TABLE `message` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `username` varchar(50) NOT NULL DEFAULT '匿名用户',
  `content` text NOT NULL,
  `contact` varchar(50) DEFAULT NULL,
  `create_time` datetime NOT NULL,
  `status` tinyint(4) NOT NULL DEFAULT '0' COMMENT '0-未处理，1-已回复',
  `type` varchar(20) DEFAULT '其他' COMMENT '咨询/建议/投诉/其他',
  `is_risk` tinyint(4) NOT NULL DEFAULT '0' COMMENT '0-正常，1-违规',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 回复表
CREATE TABLE `reply` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `message_id` int(11) NOT NULL,
  `reply_content` text NOT NULL,
  `reply_time` datetime NOT NULL,
  `admin_name` varchar(50) NOT NULL,
  PRIMARY KEY (`id`),
  KEY `fk_message_id` (`message_id`),
  CONSTRAINT `fk_message_id` FOREIGN KEY (`message_id`) REFERENCES `message` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 管理员表
CREATE TABLE `admin` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `username` varchar(30) NOT NULL UNIQUE,
  `password` varchar(30) NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

package com.example.messageboard.mapper;

import com.example.messageboard.entity.Message;
import org.apache.ibatis.annotations.Mapper;
import org.apache.ibatis.annotations.Param;
import java.util.List;

@Mapper
public interface MessageMapper {
    // 新增留言
    int insert(Message message);
    // 分页查询留言（倒序，可按状态筛选）
    List<Message> selectByPage(@Param("status") Integer status);
    // 根据ID查询留言
    Message selectById(Integer id);
    // 更新留言状态
    int updateStatus(@Param("id") Integer id, @Param("status") Integer status);
    // 更新留言类型
    int updateType(@Param("id") Integer id, @Param("type") String type);
    // 更新违规标记
    int updateRisk(@Param("id") Integer id, @Param("isRisk") Integer isRisk);
    // 删除留言（级联删除回复由数据库外键完成）
    int deleteById(Integer id);
    // 批量删除
    int deleteByIds(@Param("ids") List<Integer> ids);
    // 统计数量
    long countByStatus(@Param("status") Integer status);
    // 按类型统计
    long countByType(@Param("type") String type);
    // 统计总留言数
    long countTotal();
}

<mapper namespace="com.example.messageboard.mapper.MessageMapper">
    
    <resultMap id="MessageMap" type="com.example.messageboard.entity.Message">
        <id column="id" property="id"/>
        <result column="username" property="username"/>
        <result column="content" property="content"/>
        <result column="contact" property="contact"/>
        <result column="create_time" property="createTime"/>
        <result column="status" property="status"/>
        <result column="type" property="type"/>
        <result column="is_risk" property="isRisk"/>
    </resultMap>

    <insert id="insert" useGeneratedKeys="true" keyProperty="id">
        INSERT INTO message(username, content, contact, create_time, status, type, is_risk)
        VALUES(#{username}, #{content}, #{contact}, #{createTime}, #{status}, #{type}, #{isRisk})
    </insert>

    <select id="selectByPage" resultMap="MessageMap">
        SELECT * FROM message
        <where>
            <if test="status != null">
                status = #{status}
            </if>
        </where>
        ORDER BY create_time DESC
    </select>

    <select id="selectById" resultMap="MessageMap">
        SELECT * FROM message WHERE id = #{id}
    </select>

    <update id="updateStatus">
        UPDATE message SET status = #{status} WHERE id = #{id}
    </update>

    <update id="updateType">
        UPDATE message SET type = #{type} WHERE id = #{id}
    </update>

    <update id="updateRisk">
        UPDATE message SET is_risk = #{isRisk} WHERE id = #{id}
    </update>

    <delete id="deleteById">
        DELETE FROM message WHERE id = #{id}
    </delete>

    <delete id="deleteByIds">
        DELETE FROM message WHERE id IN
        <foreach collection="ids" item="id" open="(" separator="," close=")">

        </foreach>
    </delete>

    <select id="countByStatus" resultType="long">

@Component
public class AISimulatorUtil {
    
    // 违规词库（简单模拟）
    private static final Set<String> RISK_WORDS = new HashSet<>(Arrays.asList(
        "色情", "暴力", "反动", "赌博", "诈骗", "毒品", "恐怖", "骂人", "傻逼", "操你妈"
    ));
    
    // 分类关键词映射
    private static final Map<String, List<String>> TYPE_KEYWORDS = new HashMap<>();
    static {
        TYPE_KEYWORDS.put("咨询", Arrays.asList("请问", "如何", "怎么", "什么", "帮助", "问题"));
        TYPE_KEYWORDS.put("建议", Arrays.asList("建议", "希望", "增加", "改进", "优化", "最好"));
        TYPE_KEYWORDS.put("投诉", Arrays.asList("差评", "投诉", "不满", "糟糕", "故障", "错误", "打不开"));
        TYPE_KEYWORDS.put("其他", Arrays.asList("其他", "闲聊", "打招呼"));
    }
    
    // 回复模板
    private static final Map<String, String> REPLY_TEMPLATES = new HashMap<>();
    static {
        REPLY_TEMPLATES.put("咨询", "感谢您的咨询！我们已收到您的问题，会尽快处理并回复您。如需进一步帮助，请联系客服。");
        REPLY_TEMPLATES.put("建议", "非常感谢您提出的宝贵建议！我们会认真评估并在后续版本中优化改进。");
        REPLY_TEMPLATES.put("投诉", "非常抱歉给您带来了不好的体验！我们会立即核实并处理您反馈的问题，感谢您的监督。");
        REPLY_TEMPLATES.put("其他", "您好，您的留言我们已经收到，感谢您的支持！");
    }
    
    /**
     * 留言内容润色（模拟：去除多余空格、替换常见口语为书面语）
     */
    public static String polishContent(String content) {
        if (content == null || content.trim().isEmpty()) return "";
        // 简单润色规则：去除首尾空格，替换多个空格为单个，替换一些口语词
        String polished = content.trim().replaceAll("\\s+", " ");
        polished = polished.replace("啥", "什么")
                           .replace("咋", "怎么")
                           .replace("搞", "处理")
                           .replace("牛逼", "优秀");
        // 添加礼貌后缀（如果原文不含问号/句号）
        if (!polished.endsWith("？") && !polished.endsWith("!") && !polished.endsWith("。")) {
            polished += "。";
        }
        return polished;
    }
    
    /**
     * 检测违规内容
     * @return true表示违规
     */
    public static boolean checkRisk(String content) {
        if (content == null) return false;
        for (String word : RISK_WORDS) {
            if (content.contains(word)) {
                return true;
            }
        }
        return false;
    }
    
    /**
     * 智能分类留言
     */
    public static String classifyMessage(String content) {
        for (Map.Entry<String, List<String>> entry : TYPE_KEYWORDS.entrySet()) {
            for (String kw : entry.getValue()) {
                if (content.contains(kw)) {
                    return entry.getKey();
                }
            }
        }
        return "其他";
    }
    
    /**
     * 根据内容和分类生成回复
     */
    public static String generateReply(String content, String category) {
        // 优先使用模板
        String template = REPLY_TEMPLATES.getOrDefault(category, "感谢您的留言，我们会尽快处理。");
        // 个性化拼接：摘取留言中的部分关键词

        if (category.equals("投诉") && (content.contains("打不开") || content.contains("页面"))) {

        return personalized + " 【AI自动生成，可编辑】";

import java.util.*;

@RestController
@RequestMapping("/api")
public class ApiController {
    
    @Autowired
    private MessageService messageService;
    @Autowired
    private ReplyService replyService;
    
    // ---------- 用户端留言操作 ----------
    @PostMapping("/message/add")
    public Map<String, Object> addMessage(@RequestParam(required = false) String username,
                                          @RequestParam String content,
                                          @RequestParam(required = false) String contact) {
        Map<String, Object> result = new HashMap<>();
        // 后端校验
        if (content == null || content.trim().isEmpty()) {
            result.put("success", false);
            result.put("msg", "留言内容不能为空");
            return result;
        }
        if (content.length() > 1000) {
            result.put("success", false);
            result.put("msg", "留言内容过长，请限制在1000字以内");
            return result;
        }
        Message msg = new Message();
        msg.setUsername(username == null || username.trim().isEmpty() ? "匿名用户" : username.trim());
        msg.setContent(content.trim());
        msg.setContact(contact == null ? null : contact.trim());
        // AI自动分类
        String type = AISimulatorUtil.classifyMessage(content);
        msg.setType(type);
        // 违规检测
        boolean isRisk = AISimulatorUtil.checkRisk(content);
        msg.setIsRisk(isRisk ? 1 : 0);
        messageService.addMessage(msg);
        result.put("success", true);
        result.put("msg", "留言提交成功");
        return result;
    }
    
    @GetMapping("/message/list")
    public PageInfo<Message> listMessages(@RequestParam(defaultValue = "1") Integer page,
                                          @RequestParam(defaultValue = "10") Integer size,
                                          @RequestParam(required = false) Integer status) {
        return messageService.getMessagesByPage(page, size, status);
    }
    
    @GetMapping("/message/detail")
    public Map<String, Object> getMessageDetail(@RequestParam Integer id) {
        Map<String, Object> data = new HashMap<>();
        Message msg = messageService.getMessageDetail(id);
        Reply reply = replyService.getReplyByMessageId(id);
        data.put("message", msg);
        data.put("reply", reply);
        return data;
    }
    
    // ---------- AI 功能接口 ----------
    @PostMapping("/ai/polish")
    public Map<String, String> aiPolish(@RequestBody Map<String, String> req) {
        String content = req.get("content");
        String polished = AISimulatorUtil.polishContent(content);
        Map<String, String> res = new HashMap<>();
        res.put("polished", polished);
        return res;
    }
    
    @PostMapping("/ai/checkRisk")
    public Map<String, Boolean> aiCheckRisk(@RequestBody Map<String, String> req) {
        boolean risk = AISimulatorUtil.checkRisk(req.get("content"));
        Map<String, Boolean> res = new HashMap<>();
        res.put("isRisk", risk);
        return res;
    }
    
    @PostMapping("/ai/generateReply")
    public Map<String, String> aiGenerateReply(@RequestBody Map<String, String> req) {
        String content = req.get("content");
        String category = req.get("category");
        if (category == null) category = AISimulatorUtil.classifyMessage(content);

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

  `id` int(11) NOT NULL AUTO_INCREMENT,

  PRIMARY KEY (`id`),
  KEY `fk_message_id` (`message_id`),
  CONSTRAINT `fk_message_id` FOREIGN KEY (`message_id`) REFERENCES `message` (`id`) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `admin` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `username` varchar(30) NOT NULL UNIQUE,

package com.example.messageboard.service.impl;

import com.example.messageboard.entity.Message;
import com.example.messageboard.mapper.MessageMapper;
import com.example.messageboard.service.MessageService;
import com.github.pagehelper.PageHelper;
import com.github.pagehelper.PageInfo;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.util.Date;
import java.util.List;

@Service
public class MessageServiceImpl implements MessageService {
    @Autowired
    private MessageMapper messageMapper;

    @Override
    public void addMessage(Message message) {
        message.setCreateTime(new Date());
        message.setStatus(0);
        if (message.getIsRisk() == null) message.setIsRisk(0);
        if (message.getType() == null) message.setType("其他");
        messageMapper.insert(message);
    }

    @Override
    public PageInfo<Message> getMessagesByPage(Integer pageNum, Integer pageSize, Integer status) {
        PageHelper.startPage(pageNum, pageSize);
        List<Message> list = messageMapper.selectByPage(status);
        return new PageInfo<>(list);
    }

    @Override
    public Message getMessageDetail(Integer id) {
        return messageMapper.selectById(id);
    }

    @Override
    public boolean updateMessageStatus(Integer id, Integer status) {
        return messageMapper.updateStatus(id, status) > 0;
    }

    @Override
    public boolean updateMessageType(Integer id, String type) {
        return messageMapper.updateType(id, type) > 0;
    }

    @Override
    public boolean updateMessageRisk(Integer id, Integer isRisk) {
        return messageMapper.updateRisk(id, isRisk) > 0;
    }

    @Override
    @Transactional
    public boolean deleteMessage(Integer id) {
        return messageMapper.deleteById(id) > 0;
    }

    @Override
    @Transactional
    public boolean batchDeleteMessages(List<Integer> ids) {
        return messageMapper.deleteByIds(ids) > 0;
    }

    @Override
    public long getTotalCount() {
        return messageMapper.countTotal();
    }

    @Override
    public long getUnhandledCount() {
        return messageMapper.countByStatus(0);
    }

    @Override
    public long getHandledCount() {
        return messageMapper.countByStatus(1);
    }

    @Override
    public long getCountByType(String type) {
        return messageMapper.countByType(type);
    }
}

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import java.util.Date;

@Service
public class ReplyServiceImpl implements ReplyService {
    @Autowired
    private ReplyMapper replyMapper;

    @Override
    public Reply getReplyByMessageId(Integer messageId) {
        return replyMapper.selectByMessageId(messageId);
    }

    @Override
    public boolean saveOrUpdateReply(Reply reply) {
        reply.setReplyTime(new Date());
        Reply exist = replyMapper.selectByMessageId(reply.getMessageId());
        if (exist == null) {
            return replyMapper.insert(reply) > 0;
        } else {
            return replyMapper.update(reply) > 0;
        }
    }

    @Override
    public boolean deleteReply(Integer messageId) {
        return replyMapper.deleteByMessageId(messageId) > 0;
    }
}

package com.example.messageboard.util;

import org.springframework.stereotype.Component;
import java.util.*;

@Component
public class AISimulatorUtil {
    private static final Set<String> RISK_WORDS = new HashSet<>(Arrays.asList(
        "色情", "暴力", "反动", "赌博", "诈骗", "毒品", "恐怖", "骂人", "傻逼", "操你妈"
    ));
    
    private static final Map<String, List<String>> TYPE_KEYWORDS = new HashMap<>();
    static {
        TYPE_KEYWORDS.put("咨询", Arrays.asList("请问", "如何", "怎么", "什么", "帮助", "问题"));
        TYPE_KEYWORDS.put("建议", Arrays.asList("建议", "希望", "增加", "改进", "优化", "最好"));
        TYPE_KEYWORDS.put("投诉", Arrays.asList("差评", "投诉", "不满", "糟糕", "故障", "错误", "打不开"));
        TYPE_KEYWORDS.put("其他", Arrays.asList("其他", "闲聊", "打招呼"));
    }
    
    private static final Map<String, String> REPLY_TEMPLATES = new HashMap<>();
    static {
        REPLY_TEMPLATES.put("咨询", "感谢您的咨询！我们已收到您的问题，会尽快处理并回复您。如需进一步帮助，请联系客服。");
        REPLY_TEMPLATES.put("建议", "非常感谢您提出的宝贵建议！我们会认真评估并在后续版本中优化改进。");
        REPLY_TEMPLATES.put("投诉", "非常抱歉给您带来了不好的体验！我们会立即核实并处理您反馈的问题，感谢您的监督。");
        REPLY_TEMPLATES.put("其他", "您好，您的留言我们已经收到，感谢您的支持！");
    }
    
    public static String polishContent(String content) {
        if (content == null || content.trim().isEmpty()) return "";
        String polished = content.trim().replaceAll("\\s+", " ");
        polished = polished.replace("啥", "什么").replace("咋", "怎么").replace("搞", "处理").replace("牛逼", "优秀");
        if (!polished.endsWith("？") && !polished.endsWith("!") && !polished.endsWith("。")) {
            polished += "。";
        }
        return polished;
    }
    
    public static boolean checkRisk(String content) {
        if (content == null) return false;
        for (String word : RISK_WORDS) {
            if (content.contains(word)) return true;
        }
        return false;
    }
    
    public static String classifyMessage(String content) {
        for (Map.Entry<String, List<String>> entry : TYPE_KEYWORDS.entrySet()) {
            for (String kw : entry.getValue()) {
                if (content.contains(kw)) return entry.getKey();
            }
        }
        return "其他";
    }
    
    public static String generateReply(String content, String category) {
        String template = REPLY_TEMPLATES.getOrDefault(category, "感谢您的留言，我们会尽快处理。");
        if (category.equals("咨询") && content.contains("多久")) {
            return "感谢咨询！一般情况下我们会在24小时内回复您。";
        }
        if (category.equals("投诉") && (content.contains("打不开") || content.contains("页面"))) {
            return "抱歉给您造成访问不便！请提供更多信息（如浏览器、网络环境），我们将立刻检查优化。";
        }
        return template + " 【AI自动生成，可编辑】";
    }

package com.example.messageboard.util;

import org.springframework.web.servlet.HandlerInterceptor;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;
import javax.servlet.http.HttpSession;

public class AdminInterceptor implements HandlerInterceptor {
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception {
        HttpSession session = request.getSession();
        Object admin = session.getAttribute("admin");
        if (admin == null) {
            response.sendRedirect(request.getContextPath() + "/admin/login");
            return false;
        }
        return true;
    }
}

package com.example.messageboard;

@MapperScan("com.example.messageboard.mapper")
public class MessageBoardApplication implements WebMvcConfigurer {
    public static void main(String[] args) {
        SpringApplication.run(MessageBoardApplication.class, args);

    public void addInterceptors(InterceptorRegistry registry) {

    }
}

package com.example.messageboard.controller;

import com.example.messageboard.entity.Admin;
import com.example.messageboard.entity.Message;
import com.example.messageboard.entity.Reply;
import com.example.messageboard.service.AdminService;
import com.example.messageboard.service.MessageService;
import com.example.messageboard.service.ReplyService;
import com.example.messageboard.util.AISimulatorUtil;
import com.github.pagehelper.PageInfo;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;
import javax.servlet.http.HttpSession;
import java.util.*;

@RestController
@RequestMapping("/api")
public class ApiController {
    
    @Autowired
    private MessageService messageService;
    @Autowired
    private ReplyService replyService;
    @Autowired
    private AdminService adminService;
    
    // 用户提交留言
    @PostMapping("/message/add")
    public Map<String, Object> addMessage(@RequestParam(required = false) String username,
                                          @RequestParam String content,
                                          @RequestParam(required = false) String contact) {
        Map<String, Object> result = new HashMap<>();
        if (content == null || content.trim().isEmpty()) {
            result.put("success", false);
            result.put("msg", "留言内容不能为空");
            return result;
        }
        if (content.length() > 1000) {
            result.put("success", false);
            result.put("msg", "留言内容过长");
            return result;
        }
        Message msg = new Message();
        msg.setUsername(username == null || username.trim().isEmpty() ? "匿名用户" : username.trim());
        msg.setContent(content.trim());
        msg.setContact(contact == null ? null : contact.trim());
        msg.setType(AISimulatorUtil.classifyMessage(content));
        msg.setIsRisk(AISimulatorUtil.checkRisk(content) ? 1 : 0);
        messageService.addMessage(msg);
        result.put("success", true);
        result.put("msg", "留言提交成功");
        return result;
    }
    
    // 获取留言列表（分页）
    @GetMapping("/message/list")
    public PageInfo<Message> listMessages(@RequestParam(defaultValue = "1") Integer page,
                                          @RequestParam(defaultValue = "10") Integer size,
                                          @RequestParam(required = false) Integer status) {
        return messageService.getMessagesByPage(page, size, status);
    }
    
    // 获取留言详情及回复
    @GetMapping("/message/detail")
    public Map<String, Object> getMessageDetail(@RequestParam Integer id) {
        Map<String, Object> data = new HashMap<>();
        Message msg = messageService.getMessageDetail(id);
        Reply reply = replyService.getReplyByMessageId(id);
        data.put("message", msg);
        data.put("reply", reply);
        return data;
    }
    
    // ========== AI 接口 ==========
    @PostMapping("/ai/polish")
    public Map<String, String> aiPolish(@RequestBody Map<String, String> req) {
        String polished = AISimulatorUtil.polishContent(req.get("content"));
        return Collections.singletonMap("polished", polished);
    }
    
    @PostMapping("/ai/checkRisk")
    public Map<String, Boolean> aiCheckRisk(@RequestBody Map<String, String> req) {
        boolean risk = AISimulatorUtil.checkRisk(req.get("content"));
        return Collections.singletonMap("isRisk", risk);
    }
    
    @PostMapping("/ai/classify")
    public Map<String, String> aiClassify(@RequestBody Map<String, String> req) {
        String type = AISimulatorUtil.classifyMessage(req.get("content"));
        return Collections.singletonMap("type", type);
    }
    
    @PostMapping("/ai/generateReply")
    public Map<String, String> aiGenerateReply(@RequestBody Map<String, String> req) {
        String content = req.get("content");
        String category = req.getOrDefault("category", AISimulatorUtil.classifyMessage(content));
        String reply = AISimulatorUtil.generateReply(content, category);
        return Collections.singletonMap("reply", reply);
    }
    
    // ========== 管理员后台接口（需要登录）==========
    @PostMapping("/admin/login")
    public Map<String, Object> adminLogin(@RequestParam String username,
                                          @RequestParam String password,
                                          HttpSession session) {
        Map<String, Object> result = new HashMap<>();
        Admin admin = adminService.login(username, password);
        if (admin != null) {
            session.setAttribute("admin", admin);
            result.put("success", true);
            result.put("msg", "登录成功");
        } else {
            result.put("success", false);
            result.put("msg", "用户名或密码错误");
        }
        return result;
    }
    
    @GetMapping("/admin/logout")
    public Map<String, Object> adminLogout(HttpSession session) {
        session.removeAttribute("admin");
        Map<String, Object> result = new HashMap<>();
        result.put("success", true);
        return result;
    }
    
    @PostMapping("/admin/message/delete")
    public Map<String, Object> deleteMessage(@RequestParam Integer id) {
        boolean success = messageService.deleteMessage(id);
        Map<String, Object> result = new HashMap<>();
        result.put("success", success);
        return result;
    }
    
    @PostMapping("/admin/message/batchDelete")
    public Map<String, Object> batchDeleteMessages(@RequestBody List<Integer> ids) {
        boolean success = messageService.batchDeleteMessages(ids);
        Map<String, Object> result = new HashMap<>();
        result.put("success", success);
        return result;
    }
    
    @PostMapping("/admin/message/updateStatus")
    public Map<String, Object> updateMessageStatus(@RequestParam Integer id, @RequestParam Integer status) {
        boolean success = messageService.updateMessageStatus(id, status);
        Map<String, Object> result = new HashMap<>();
        result.put("success", success);
        return result;
    }
    
    @PostMapping("/admin/message/updateType")
    public Map<String, Object> updateMessageType(@RequestParam Integer id, @RequestParam String type) {
        boolean success = messageService.updateMessageType(id, type);
        Map<String, Object> result = new HashMap<>();
        result.put("success", success);
        return result;
    }
    
    @PostMapping("/admin/reply/save")
    public Map<String, Object> saveReply(@RequestParam Integer messageId,
                                         @RequestParam String replyContent,
                                         HttpSession session) {
        Admin admin = (Admin) session.getAttribute("admin");
        Reply reply = new Reply();
        reply.setMessageId(messageId);
        reply.setReplyContent(replyContent);
        reply.setAdminName(admin.getUsername());
        boolean success = replyService.saveOrUpdateReply(reply);
        if (success) {
            // 更新留言状态为已回复
            messageService.updateMessageStatus(messageId, 1);
        }
        Map<String, Object> result = new HashMap<>();
        result.put("success", success);
        return result;
    }
    
    // AI 批量检测所有留言违规
    @PostMapping("/admin/ai/batchCheckRisk")
    public Map<String, Object> batchCheckRisk() {
        PageInfo<Message> pageInfo = messageService.getMessagesByPage(1, Integer.MAX_VALUE, null);
        List<Message> allMessages = pageInfo.getList();
        int riskCount = 0;
        for (Message msg : allMessages) {
            boolean isRisk = AISimulatorUtil.checkRisk(msg.getContent());
            int newRisk = isRisk ? 1 : 0;
            if (msg.getIsRisk() != newRisk) {
                messageService.updateMessageRisk(msg.getId(), newRisk);
            }
            if (isRisk) riskCount++;
        }
        Map<String, Object> result = new HashMap<>();
        result.put("success", true);
        result.put("riskCount", riskCount);
        result.put("msg", "检测完成，共发现 " + riskCount + " 条违规留言");
        return result;
    }
    
    // 获取统计数据（用于AI统计面板）
    @GetMapping("/admin/stats/data")
    public Map<String, Object> getStatsData() {
        Map<String, Object> data = new HashMap<>();
        data.put("total", messageService.getTotalCount());
        data.put("unhandled", messageService.getUnhandledCount());
        data.put("handled", messageService.getHandledCount());
        Map<String, Long> typeCounts = new HashMap<>();
        typeCounts.put("咨询", messageService.getCountByType("咨询"));
        typeCounts.put("建议", messageService.getCountByType("建议"));
        typeCounts.put("投诉", messageService.getCountByType("投诉"));
        typeCounts.put("其他", messageService.getCountByType("其他"));
        data.put("typeCounts", typeCounts);
        return data;
    }
}

    <script>
        let currentPage = 1;
        function loadMessages(page) {
            currentPage = page;
            fetch(`/api/message/list?page=${page}&size=5`)
                .then(res => res.json())
                .then(data => {
                    renderMessages(data.list);
                    renderPagination(data);
                });
        }
        function renderMessages(messages) {
            const container = document.getElementById('messageList');
            if (!messages.length) {
                container.innerHTML = '<p>暂无留言</p>';
                return;
            }
            let html = '';
            messages.forEach(msg => {
                let riskClass = msg.isRisk === 1 ? 'risk' : '';
                html += `<div class="message-card ${riskClass}">
                            <div><strong>${escapeHtml(msg.username)}</strong> 发表于 ${msg.createTime}</div>
                            <div>${escapeHtml(msg.content.substring(0, 100))}${msg.content.length>100?'...':''}</div>
                            <div>状态：${msg.status === 0 ? '未处理' : '已回复'} | 分类：${msg.type}</div>
                            <a href="/detail?id=${msg.id}">查看详情</a>
                         </div>`;
            });
            container.innerHTML = html;
        }
        function renderPagination(pageInfo) {
            let pagination = document.getElementById('pagination');
            let html = '<div class="pagination">';
            if (pageInfo.hasPreviousPage) {
                html += `<button onclick="loadMessages(${pageInfo.prePage})">上一页</button>`;
            }
            html += `<span>第 ${pageInfo.pageNum} / ${pageInfo.pages} 页</span>`;
            if (pageInfo.hasNextPage) {
                html += `<button onclick="loadMessages(${pageInfo.nextPage})">下一页</button>`;
            }
            html += '</div>';
            pagination.innerHTML = html;
        }
        function fillTemplate(type) {
            let text = '';
            if(type === '咨询') text = '请问……？希望能得到解答。';
            else if(type === '建议') text = '我建议……这样能提升用户体验。';
            else if(type === '感谢') text = '非常感谢开发团队，系统很好用！';
            document.getElementById('content').value = text;
        }
        function polishContent() {
            let content = document.getElementById('content').value;
            fetch('/api/ai/polish', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({content: content})
            }).then(res => res.json())
              .then(data => { document.getElementById('content').value = data.polished; });
        }
        function checkRisk() {
            let content = document.getElementById('content').value;
            fetch('/api/ai/checkRisk', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({content: content})
            }).then(res => res.json())
              .then(data => {
                  if(data.isRisk) alert('⚠️ 检测到违规内容，请修改后再提交！');
                  else alert('✅ 内容合法，可以提交。');
              });
        }
        function submitMessage() {
            let username = document.getElementById('username').value;
            let content = document.getElementById('content').value;
            let contact = document.getElementById('contact').value;
            if(!content.trim()) {
                alert('留言内容不能为空');
                return;
            }
            let params = new URLSearchParams();
            params.append('username', username);
            params.append('content', content);
            params.append('contact', contact);
            fetch('/api/message/add', {
                method: 'POST',
                headers: {'Content-Type': 'application/x-www-form-urlencoded'},
                body: params
            }).then(res => res.json())
              .then(data => {
                  alert(data.msg);
                  if(data.success) {
                      document.getElementById('content').value = '';
                      loadMessages(1);
                  }
              });
        }
        function escapeHtml(str) {
            return str.replace(/[&<>]/g, function(m) {
                if(m === '&') return '&amp;';
                if(m === '<') return '&lt;';
                if(m === '>') return '&gt;';
                return m;
            });
        }
        loadMessages(1);
    </script>

        let selectedIds = [];
        function loadMessages(page) {
            let status = document.getElementById('statusFilter').value;
            let url = `/api/message/list?page=${page}&size=10`;
            if(status !== '') url += `&status=${status}`;
            fetch(url).then(res => res.json()).then(data => {

        function renderMessages(messages) {
            let html = '<table border="1" cellpadding="8"><tr><th><input type="checkbox" onclick="toggleAll(this)"></th><th>ID</th><th>留言人</th><th>内容摘要</th><th>分类</th><th>状态</th><th>违规</th><th>操作</th></tr>';
            messages.forEach(msg => {
                html += `<tr>
                            <td><input type="checkbox" class="msg-check" value="${msg.id}"></td>
                            <td>${msg.id}</td>
                            <td>${escapeHtml(msg.username)}</td>
                            <td>${escapeHtml(msg.content.substring(0,50))}</td>
                            <td><select onchange="updateType(${msg.id}, this.value)"><option value="咨询" ${msg.type=='咨询'?'selected':''}>咨询</option><option value="建议" ${msg.type=='建议'?'selected':''}>建议</option><option value="投诉" ${msg.type=='投诉'?'selected':''}>投诉</option><option value="其他" ${msg.type=='其他'?'selected':''}>其他</option></select></td>
                            <td><button onclick="toggleStatus(${msg.id}, ${msg.status})">${msg.status==0?'标记已回复':'标记未处理'}</button></td>
                            <td>${msg.isRisk==1?'⚠️违规':'正常'}</td>
                            <td><button onclick="deleteMsg(${msg.id})">删除</button> | <button onclick="aiGenerateReply(${msg.id})">AI生成回复</button></td>
                         </tr>`;
            });
            html += '</table>';
            document.getElementById('messageList').innerHTML = html;
            // 绑定复选框事件
            document.querySelectorAll('.msg-check').forEach(cb => {
                cb.addEventListener('change', function() {
                    if(this.checked) selectedIds.push(parseInt(this.value));
                    else selectedIds = selectedIds.filter(id => id !== parseInt(this.value));
                });
            });
        }
        function toggleAll(checkbox) {
            document.querySelectorAll('.msg-check').forEach(cb => cb.checked = checkbox.checked);
            selectedIds = checkbox.checked ? Array.from(document.querySelectorAll('.msg-check')).map(cb => parseInt(cb.value)) : [];
        }
        function batchDelete() {
            if(selectedIds.length === 0) { alert('请选择留言'); return; }
            if(confirm(`确定删除 ${selectedIds.length} 条留言吗？`)) {
                fetch('/api/admin/message/batchDelete', {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify(selectedIds)

        function toggleStatus(id, currentStatus) {

            fetch(`/api/admin/message/updateStatus?id=${id}&status=${newStatus}`, {method:'POST'})
                .then(res => res.json())
                .then(data => { if(data.success) loadMessages(1); });
        }
        function updateType(id, type) {
            fetch(`/api/admin/message/updateType?id=${id}&type=${type}`, {method:'POST'})
                .then(res => res.json())
                .then(data => { if(data.success) alert('分类更新成功'); });
        }
        function batchCheckRisk() {
            fetch('/api/admin/ai/batchCheckRisk', {method:'POST'})
                .then(res => res.json())
                .then(data => { alert(data.msg); loadMessages(1); });
        }
        function aiGenerateReply(id) {
            // 先获取留言内容
            fetch(`/api/message/detail?id=${id}`).then(res => res.json()).then(data => {

                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({content: content})

                        params.append('messageId', id);
                        params.append('replyContent', reply);

                            .then(res => res.json())
                            .then(r => { if(r.success) alert('回复保存成功'); loadMessages(1); });
                    }
                });
            });
        }

package com.example.messageboard;

import com.example.messageboard.util.AdminInterceptor;
import org.mybatis.spring.annotation.MapperScan;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.web.servlet.config.annotation.InterceptorRegistry;
import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;

@SpringBootApplication
@MapperScan("com.example.messageboard.mapper")
public class MessageBoardApplication implements WebMvcConfigurer {
    public static void main(String[] args) {
        SpringApplication.run(MessageBoardApplication.class, args);
    }
    
    @Override
    public void addInterceptors(InterceptorRegistry registry) {
        registry.addInterceptor(new AdminInterceptor())
                .addPathPatterns("/admin/**")
                .excludePathPatterns("/admin/login", "/api/admin/login", "/css/**", "/js/**");

# 在线留言板管理系统 - 课程设计

## 环境要求
- JDK 1.8+
- MySQL 5.7+
- Maven 3.6+

## 快速启动
1. 创建数据库 `message_board_db`，执行 `src/sql/schema.sql`
2. 修改 `application.properties` 中的数据库用户名/密码
3. 运行 `mvn spring-boot:run` 或在 IDE 中启动主类
4. 访问：
   - 用户端：http://localhost:8080/
   - 管理员登录：http://localhost:8080/admin/login （账号 admin / 123456）

## AI 功能说明
- 所有 AI 功能均采用本地模拟算法，无需网络即可运行。
- 如需对接阿里云百炼大模型，请取消 `AISimulatorUtil` 中注释并配置 API Key。
